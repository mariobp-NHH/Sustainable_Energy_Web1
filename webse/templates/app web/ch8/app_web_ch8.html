script
    Chart.scaleService.updateScaleDefaults('linear', {
            ticks: {
                min: 0
            }
    });
    let co2_transport_chart = new Chart(co2_transport, {
            type: 'pie',
            data: {
            labels: ['CO2 bus'],
                datasets: [{
                    label: "CO2 transport",
                    data: [{{co2_bus}}],
                    backgroundColor: ['#f2935c'],

                    borderWidth: 1,
                    hoverBorderColor: "black",
                    hoverBorderWidth: 2,
                    hoverBackgroundColor: 'rgba(154, 245, 140)',
                    pointHoverRadius: 5
                }],
            },
            options: {
                    title: {
                        display: true,
                            text: "CO2 Transport",
                                fontSize: 20,
                    },
                    legend: {
                        position: "right",
                            labels: {
                            fontColor: "gray"
                        },
                        display: true,
                    },

                    elements: {
                        hitRadius: 3,
                    }
            }
        })
